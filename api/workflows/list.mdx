---
title: List Workflows
api: GET /workflows
---

Retrieves a paginated list of workflows for the authenticated user with optional filtering.

## Query Parameters

<ParamField query="cursor" type="string">
  Pagination cursor for fetching the next page of results
</ParamField>

<ParamField query="query" type="string">
  Search query string for filtering workflows by name
</ParamField>

<ParamField query="kind" type="string">
  Filter workflows by kind
</ParamField>

<ParamField query="build_status" type="string">
  Filter workflows by build status. Cannot be used when `terminated=true`
</ParamField>

<ParamField query="terminated" type="boolean" default="false">
  Filter for terminated workflows. Set to `true` to show only terminated workflows
</ParamField>

<ParamField query="interval_min" type="integer">
  Minimum interval value (in minutes) for filtering workflows
</ParamField>

<ParamField query="interval_max" type="integer">
  Maximum interval value (in minutes) for filtering workflows. Must be greater than or equal to `interval_min`
</ParamField>

## Response

<ResponseField name="workflows" type="array">
  <Expandable title="Workflow Object">
    <ResponseField name="id" type="string">
      Unique identifier of the workflow
    </ResponseField>

    <ResponseField name="name" type="string">
      Name of the workflow
    </ResponseField>

    <ResponseField name="payload" type="string">
      JSON string containing the workflow payload configuration
    </ResponseField>

    <ResponseField name="kind" type="string">
      Kind of workflow
    </ResponseField>

    <ResponseField name="build_status" type="string">
      Current build status of the workflow
    </ResponseField>

    <ResponseField name="interval" type="integer">
      Execution interval measured in minutes
    </ResponseField>

    <ResponseField name="consecutive_job_failures_count" type="integer">
      Current count of consecutive job failures
    </ResponseField>

    <ResponseField name="max_consecutive_job_failures_allowed" type="integer">
      Maximum number of consecutive job failures allowed
    </ResponseField>

    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp when the workflow was created
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp when the workflow was last updated
    </ResponseField>

    <ResponseField name="terminated_at" type="string">
      ISO 8601 timestamp when the workflow was terminated (if applicable)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="cursor" type="string">
  Pagination cursor for fetching the next page of results. Empty string indicates no more pages
</ResponseField>

## Example Request

```bash
curl -X GET "https://api.chronoverse.com/workflows?kind=scheduled&terminated=false&interval_min=60&interval_max=1440" \
  -H "Authorization: Bearer YOUR_API_TOKEN"
```

## Pagination

To retrieve the next page of results, use the `cursor` value from the response in your next request:

```bash
curl -X GET "https://api.chronoverse.com/workflows?cursor=eyJpZCI6IndmXzk4NzY1NDMyMTBmZWRjYmEifQ==" \
  -H "Authorization: Bearer YOUR_API_TOKEN"
```
